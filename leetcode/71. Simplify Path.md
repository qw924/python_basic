class Solution:
    def simplifyPath(self, path):
        """
        :type path: str
        :rtype: str
        """
        stack = []
        for p in path.split('/'):
            if p:
                if p == '..':
                    if stack:
                        stack.pop()
                elif p!='.':
                    stack.append(p)
                
        return '/' + '/'.join(stack)
